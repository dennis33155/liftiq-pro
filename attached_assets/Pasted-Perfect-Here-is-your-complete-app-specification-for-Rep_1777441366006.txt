PROGRESSIVE OVERLOAD LOGIC

func suggestNextWeight(lastWeight: Double, lastReps: Int) -> String {
    let nextWeight = lastWeight + 2.5
    let nextRepsLow = lastReps
    let nextRepsHigh = lastReps + 1
    return "\(nextWeight) lb x \(nextRepsLow)-\(nextRepsHigh)"
}


Show this suggestion in green on every exercise card below the last performance.

PR DETECTION LOGIC

func isNewPR(exercise: String, weight: Double, reps: Int) -> Bool {
    guard let current = personalBests[exercise] else { return true }
    if weight > current.weight { return true }
    if weight == current.weight && reps > current.reps { return true }
    return false
}
